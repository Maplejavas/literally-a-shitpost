Shinies
	*luckypipi
		on click:log Woop
		movement:onRight moveLeft fade bounce:0.05
		frequency:60
		frequency variation:30
		icon:https://cdn.discordapp.com/attachments/555753111348576264/908447351436349550/859003798453223424.gif
		class:bigButton
		on click:
			$boost=1
			if (have clover) $boost=4
			if (chance(25%))
				//get at least 7, or between 1 and 3 minutes of our pipi production
				$amount=max(7,random(pipis:ps*60*1,pipis:ps*60*3))*$boost
				toast The lucky pipi grants you<//><b>[$amount] pipis</b>.
				yield $amount pipis
			else
				$amount=1*$boost
				toast The lucky pipi grants you<//><b>[$amount] meseta[s?$amount]</b>!
				yield $amount meseta
			end
		end
